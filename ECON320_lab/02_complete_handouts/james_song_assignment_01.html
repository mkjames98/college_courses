def gen_mean_n_flip(birthday_string: str, rng: np.random.mtrand.RandomState, n: int) -> np.float64:
    cointoss = rng.choice(2, n) # takes n flips as input
    # 2 is the result (1 or 0) for cointoss
    # 10 is the amount of times this process is repeated.
    return np.mean(cointoss)

def gen_mean_1k_trials(birthday_string, rng, n) -> list:
    # list comprehension
    # mean of tossing coin toss trial (1 trial = cointoss 10 times) x 1000 saved as list
    thousand_means = [gen_mean_n_flip(birthday_string, rng, n) for _ in range(0,1000)]
    return thousand_means
